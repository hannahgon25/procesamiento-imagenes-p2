üìπ Captura de Video con OpenCV

Se usa el m√≥dulo cv2.VideoCapture() para acceder a c√°maras web o archivos de video. Permite manipular cada frame en tiempo real, aplicando filtros o an√°lisis morfol√≥gico.

cap = cv.VideoCapture(0)
if not cap.isOpened():
    raise Exception("ERROR! No se pudo abrir la c√°mara")

is_gray = False
is_blurred = False

while True:
    _, frame = cap.read()

    if is_gray:
        frame = cv.cvtColor(frame, cv.COLOR_BGR2GRAY)
    if is_blurred:
        frame = cv.blur(frame, (11, 11))

    key = cv.waitKey(17)
    if key == ord("q"):
        exit()
    elif key == ord("g"):
        is_gray = not is_gray
    elif key == ord("b"):
        is_blurred = not is_blurred

    cv.imshow("frame", frame)

cap.release()
cv.destroyAllWindows()
